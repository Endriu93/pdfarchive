<h3>A demonstration of how to access a File Upload Button</h3>

<input type="file" id="myFile">

<p>Click the "Try it" button to disable the file upload button.</p>

<button onclick="myFunction();">Try it</button>

<script>
var myFun = (function() {
      var isEnabled = true;
      var innerFun = function() { 
         var x = document.getElementById("myFile");
          if(isEnabled) {x.disabled = true; isEnabled = false;}
          else {x.disabled = false; isEnabled = true;}
     }
     return innerFun; 

})();
function myFunction() {
    var x = document.getElementById("myFile");
    alert(x.files[0].name);
}
</script>
